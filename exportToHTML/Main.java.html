public class Main {
    public static int[][] fun(int[][] matrix) {
        int a = matrix.length;
        int[][] rotated = new int[a][a];
        for (int i = 0; i < a; i++) {
            for (int j = 0; j < a; j++) {
                rotated[j][a - 1 - i] = matrix[i][j];
            }
        }
        for (int i = 0; i < a; i++) {
            if (rotated[i][i] % 2 != 0) {
                rotated[i][i] = 0;
            }
        }
        return rotated;
    }
    public static void printMatrix(int[][] matrix) {
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[i].length; j++) {
                System.out.print(matrix[i][j]);
            }
            System.out.println();
        }
    }
    public static void main(String[] args) {
        int[][] matrix = {
                {1, 2, 3},
                {4, 5, 6},
                {7, 8, 9}
        };
        System.out.println("Original Matrix:");
        printMatrix(matrix);
        int[][] rotated = fun(matrix);
        System.out.println();
        System.out.println("Rotated Matrix (Clockwise):");
        printMatrix(rotated);
    }
}
